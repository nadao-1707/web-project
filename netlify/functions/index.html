Serverless Functions Example

Hello Function

This is the hello.js function:

exports.handler = async (event, context) => {
    return {
        statusCode: 200,
        body: JSON.stringify({ message: 'Hello, this is a serverless function!' }),
    };
};

Greet Function

This is the greet.js function using Express:

const express = require('express');
const app = express();

app.get('/greet/:name', (req, res) => {
    const { name } = req.params;
    res.json({ message: `Hello, ${name}! This is a serverless function using Express!` });
});

module.exports = app;
